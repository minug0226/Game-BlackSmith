# 포인터 마무리

```cpp
#include <iostream>
    using namespace std;

    // 오늘의 주제 : 포인터 마무리

    // 1) 포인터 vs 배열
    // 2) 주의 사항 (마음가짐?)

    int main() 
    {
        // 주소를 담는 바구니
        // 진퉁은 저~ 멀리 어딘가에 있음
        // p는 단지 그곳으로 워프하는 포탈
        int* p;
        
        // 진짜배기! 원조데이터
        // 닭장처럼 데이터의 묶음 (엄청 많고 거대함)
        int arr[10] = { 1, 2, 3, 4, 5, 6, 7, 8 };

        // 그런데 상당히 많은 사람들이 [배열 = 포인터]라 착각하는 경향이 있음!
        
        // - [배열의 이름]은 배열의 시작 주소값을 가리키는 TYPE* 포인터로 쉽게 변환이 가능하기 때문인것 같다!
        p = arr;
        // - [TYPE형 1차원 배열] 과 [TYPE*형 포인터]는 완전히 호환이 된다.
        cout << p[0] << endl;
        cout << arr[0] << endl;
        cout << p[5] << endl;
        cout << arr[5] << endl;

        cout << *p << endl; // p[0]
        cout << *arr << endl; // arr[0]
        cout << *(p + 3) << endl;
        cout << *(arr + 3) << endl;

        // - 2차원 배열과 다중 포인터는 함꼐 호환이 되지않는다
        // 포인터라든지 참조를 다룰때 주소값을 다룰떄 주소값을 외부로 넘겨주는 행위는 절대로 해서는 안된다.

        return 0;
    }
```
