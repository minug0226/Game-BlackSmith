# 유의사항 ( 팁 )

```cpp
#include <iostream>
using namespace std;

// 오늘의 주제 : 유의사항 (팁)

// 1 ) 변수의 유효범위

// 전역 변수
//int hp = 10;

// 스택
// { } 중괄호의 범위가 생존 범위

// 2 ) 연산의 우선순위

// 3 ) 타입 변환

// 4 ) 사칙 연산

int main()
{

		// 1 )
		// 스택
		int hp = 777777;
		cout << hp << endl;
		

		// 2 )
		// 짝 수 여 부 
		bool isEven = ((hp % 2) == 0);
		
		// 짝수거나 3으로 나뉘는 값인지
		bool isEvenOrDivBy3 = ((hp % 2) == 0) || ((hp % 3) == 0);

		// 3 )
		// 바구니 교체
		// 위험한 행위이니 항상 주의깊게 봐야한다는 걸 명심하자.

		short hp2 = hp; // 큰 데이터 -> 작은 데이터 로 이동시 윗쪽 비트 데이터가 짤린 상태로 저장 유의
		float hp3 = hp;	 // 정수에서 사용하다가 이렇게 실수로 변환할 때 정밀도 차이가 있기 때문에 데이터 손실에 유의 
		unsigned int hp4 = hp; // 비트 단위로 보면 똑같은데, 분석하는  방법이 달라진다.

		// 4 )
		// 곱셈 
		// - 오버플로우 (너무 큰 값을 곱할때 조심)
		// 나눗셈
		// - 0 나누기 조심
		// - 실수 관련 

}
```
